class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        num_map = {}
        for index, num in enumerate(nums):
            a = target - num
            if num in num_map:
                return [num_map[num], index]
            else:
                num_map[a] = index
